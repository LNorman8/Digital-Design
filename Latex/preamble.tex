% Based on the ideas: https://tex.stackexchange.com/questions/241013/mdframed-or-shaded-inside-mdframed
\newcounter{process}
\renewcommand{\theprocess}{\thechapter.\arabic{process}}

\newenvironment{process}[1]  {
	\par%
	\addvspace{\baselineskip}	
	
	\addcontentsline{pro}{processextract}{\protect\numberline{\theprocess}#1}	
	  	
	\refstepcounter{process}
	\begin{mdframed}[
	    frametitle={ \textbf{Process \theprocess}: #1 \vspace{12pt} },
	    frametitlerule=false,
	    innertopmargin=-0.7em, innerleftmargin=0.4em,
	    hidealllines=true, leftline=true,
	    linewidth=10pt,
	    linecolor=gray!80,
	    backgroundcolor=blue!10
	]
	}{
	\end{mdframed}
}



% Based on the ideas: https://tex.stackexchange.com/questions/241013/mdframed-or-shaded-inside-mdframed
\newcounter{example}
\renewcommand{\theexample}{\thechapter.\arabic{example}}
\newenvironment{example}[1] 
{

    frametitle={ \textbf{Example \theexample} #1 \vspace{12pt} },
    frametitlerule=false,
    innertopmargin=-0.7em, innerleftmargin=0.4em,
    hidealllines=true, leftline=true,
    linewidth=10pt,
    linecolor=gray!80,
    backgroundcolor=blue!10
]%
}{%
\end{mdframed}
}
